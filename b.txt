DWORD GetRobloxPID() {
    PROCESSENTRY32 pe32; pe32.dwSize = sizeof(PROCESSENTRY32);
    HANDLE snap = CreateToolhelp32Snapshot(TH32CS_SNAPPROCESS, 0);
    if (Process32First(snap, &pe32)) {
        do {
            if (strcmp(pe32.szExeFile, "RobloxPlayerBeta.exe") == 0) {
                CloseHandle(snap);
                return pe32.th32ProcessID;
            }
        } while (Process32Next(snap, &pe32));
    }
    CloseHandle(snap);
    return 0;
}

void AimAt(HANDLE hProc, Vector3 target) {
    POINT cursor;
    GetCursorPos(&cursor);
    // Calculate delta and smooth move
    float smooth = 0.5f; // Adjust smoothness
    int screenX = (int)(target.x * smooth + cursor.x * (1 - smooth));
    int screenY = (int)(target.y * smooth + cursor.y * (1 - smooth));
    SetCursorPos(screenX, screenY);
}

int main() {
    DWORD pid = GetRobloxPID();
    if (!pid) { std::cout << "Launch Roblox Rivals first.\n"; return 1; }

    HANDLE hProc = OpenProcess(PROCESS_ALL_ACCESS, FALSE, pid);
    uintptr_t base = GetModuleBase(pid, "RobloxPlayerBeta.exe");

    std::cout << "Rivals External Aimbot Active. Hold RIGHT MOUSE to aim.\n";

    while (true) {
        if (GetAsyncKeyState(VK_RBUTTON) & 0x8000) {  // Right click hold
            auto players = GetPlayers(hProc, base);
            Player* closest = nullptr;
            float minDist = 999999.0f;

            for (auto& p : players) {
                if (p.isEnemy) {
                    Vector3 screen = WorldToScreen(hProc, p.headPos);
                    float dist = std::sqrt(std::pow(screen.x - 1920/2, 2) + std::pow(screen.y - 1080/2, 2));
                    if (dist < minDist) {
                        minDist = dist;
                        closest = &p;
                    }
                }
            }

            if (closest) AimAt(hProc, WorldToScreen(hProc, closest->headPos));
        }
        Sleep(1); // Low CPU
    }

    CloseHandle(hProc);
    return 0;
}
